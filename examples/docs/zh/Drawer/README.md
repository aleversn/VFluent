<script>
  export default {
    data(){
      return {
        visible:true,
        theme:0
      }
    },
    methods:{
      toggle(){
        this.visible^=true;
      }
    }, 
    computed:{
      $theme(){
        return !this.theme?'light':'dark';
      },
      divStyle(){
        if (this.$theme=='light'){
          return {
            backgroundColor:'#fff',
            padding:'20px',
            color:'#000',
          }
        }else{
          return {
            backgroundColor:'#000',
            padding:'20px',
            color:'#fff',
          }
        }
      },
    }
  }
</script>
